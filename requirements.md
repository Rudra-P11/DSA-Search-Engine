# Node.js Dependencies Installation Commands

npm install cors@^2.8.5
npm install express@^5.1.0
npm install natural@^8.1.0
npm install puppeteer@^24.18.0
npm install stopword@^3.1.5

# Development dependencies
npm install --save-dev nodemon@^3.1.10
